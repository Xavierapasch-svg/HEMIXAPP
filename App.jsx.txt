// ─── PERSISTENT STORAGE HELPERS ──────────────────────────────────────────────
const STORAGE_KEY = "hemovida_data_v2";

function loadData() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    if (raw) return JSON.parse(raw);
  } catch (_) {}
  return null;
}

function saveData(data) {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
    return true;
  } catch (_) {
    return false;
  }
}

                {/* INDICACIONES */}
                {procTab === "indicaciones" && (
                  <div style={S.card}>
                    <div style={S.cardTitle}>
                      <span>INDICACIONES CLÍNICAS</span>
                      {editMode && <button style={S.addBtn} onClick={() => addIndication(activeSpec, liveProc.id)}>+ Agregar</button>}
                    </div>
                    {(liveProc.indications || []).map((ind, i) => (
                      <div key={i} style={S.listRow}>
                        <span style={S.listDot}>▸</span>
                        {editMode ? (
                          <div style={{ flex: 1, display: "flex", gap: 6, alignItems: "center" }}>
                            <input style={{ ...EDIT_INPUT, flex: 1 }} value={ind} onChange={e => updateIndication(activeSpec, liveProc.id, i, e.target.value)} />
                            <button style={S.removeBtn} onClick={() => removeIndication(activeSpec, liveProc.id, i)}>✕</button>
                          </div>
                        ) : <span>{ind}</span>}
                      </div>
                    ))}
                    {(!liveProc.indications || liveProc.indications.length === 0) && (
                      <div style={{ color: C.muted, fontSize: 12, padding: "8px 0" }}>
                        {editMode ? "Haz click en '+ Agregar' para añadir indicaciones." : "Sin indicaciones registradas."}
                      </div>
                    )}
                  </div>
                )}
              </>
            );
          })()}

          {/* ══ SIDE PANELS ══ */}
          {screen === "side" && (
            <>
              {/* DICTIONARY */}
              {sideTab === "dict" && (
                <>
                  <div style={S.panelTitle}>
                    📖 DICCIONARIO MÉDICO
                    {editMode && <button style={{ ...S.addBtn, float: "right" }} onClick={addDictTerm}>+ Agregar término</button>}
                  </div>
                  <input style={{ ...S.input, marginBottom: 16 }} placeholder="Buscar término o definición..." value={dictSearch} onChange={e => setDictSearch(e.target.value)} />
                  <div style={{ fontSize: 10, color: C.muted, marginBottom: 10, letterSpacing: 1 }}>{filteredDict.length} RESULTADOS</div>
                  {data.dictionary.map((d, i) => {
                    const match = d.term.toLowerCase().includes(dictSearch.toLowerCase()) || d.def.toLowerCase().includes(dictSearch.toLowerCase());
                    if (!match) return null;
                    return (
                      <div key={i} style={S.dictItem}>
                        {editMode ? (
                          <div style={{ display: "flex", gap: 8, flexDirection: "column" }}>
                            <div style={{ display: "flex", gap: 6, alignItems: "center" }}>
                              <input style={{ ...EDIT_INPUT, flex: 1 }} value={d.term} onChange={e => updateDictTerm(i, "term", e.target.value)} />
                              <button style={S.removeBtn} onClick={() => removeDictTerm(i)}>✕</button>
                            </div>
                            <textarea style={{ ...EDIT_INPUT, minHeight: 60, resize: "vertical" }} value={d.def} onChange={e => updateDictTerm(i, "def", e.target.value)} />
                          </div>
                        ) : (
                          <>
                            <div style={S.dictTerm}>{d.term}</div>
                            <div style={S.dictDef}>{d.def}</div>
                          </>
                        )}
                      </div>
                    );
                  })}
                </>
              )}

              {/* CALCULATOR */}
              {sideTab === "calc" && (
                <>
                  <div style={S.panelTitle}>🧮 CALCULADORA DE DOSIS</div>
                  <div style={{ display: "flex", gap: 6, marginBottom: 20, flexWrap: "wrap" }}>
                    {CALCULATORS.map((c, i) => (
                      <button key={c.id} style={S.tab(calcIdx === i)} onClick={() => { setCalcIdx(i); setCalcVals({}); setCalcResult(null); }}>{c.name.toUpperCase()}</button>
                    ))}
                  </div>
                  <div style={S.calcCard}>
                    <div style={{ fontSize: 11, color: C.accent, letterSpacing: 2, marginBottom: 14 }}>{CALCULATORS[calcIdx].name.toUpperCase()}</div>
                    {CALCULATORS[calcIdx].fields.map(f => (
                      <div key={f.key} style={{ marginBottom: 12 }}>
                        <div style={{ fontSize: 11, color: C.muted, marginBottom: 5, letterSpacing: 1 }}>{f.label.toUpperCase()} ({f.unit})</div>
                        <input type="number" style={S.input} placeholder={`Ingresa ${f.label.toLowerCase()}`} value={calcVals[f.key] || ""} onChange={e => setCalcVals({ ...calcVals, [f.key]: e.target.value })} />
                      </div>
                    ))}
                    <button style={S.btn("#00d4ff")} onClick={runCalc}>CALCULAR</button>
                    {calcResult && (
                      <div style={S.resultBox(calcResult.error)}>
                        <div style={S.resultVal(calcResult.error)}>{calcResult.resultado}</div>
                        {calcResult.detalle && <div style={S.resultDetail}>{calcResult.detalle}</div>}
                      </div>
                    )}
                  </div>
                  <div style={{ ...S.card, marginTop: 12, fontSize: 11, color: C.muted, lineHeight: 1.7, borderColor: "#0a1a26" }}>
                    ⚠ Herramienta orientativa. La indicación y verificación de dosis es responsabilidad del médico tratante.
                  </div>
                </>
              )}

              {/* FAQ */}
              {sideTab === "faq" && (
                <>
                  <div style={S.panelTitle}>
                    ❓ PREGUNTAS FRECUENTES
                    {editMode && <button style={{ ...S.addBtn, float: "right" }} onClick={addFaq}>+ Agregar</button>}
                  </div>
                  {data.faqs.map((f, i) => (
                    <div key={i} style={{ ...S.card, ...S.faqCard }}>
                      {editMode ? (
                        <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>
                          <div style={{ display: "flex", gap: 6, alignItems: "flex-start" }}>
                            <input style={{ ...EDIT_INPUT, flex: 1 }} value={f.q} onChange={e => updateFaq(i, "q", e.target.value)} />
                            <button style={S.removeBtn} onClick={() => removeFaq(i)}>✕</button>
                          </div>
                          <textarea style={{ ...EDIT_INPUT, minHeight: 80, resize: "vertical" }} value={f.a} onChange={e => updateFaq(i, "a", e.target.value)} />
                        </div>
                      ) : (
                        <>
                          <div style={S.faqQ}>{f.q}</div>
                          <div style={S.faqA}>{f.a}</div>
                        </>
                      )}
                    </div>
                  ))}
                </>
              )}

              {/* SIMULATOR */}
              {sideTab === "sim" && (
                <>
                  <div style={S.panelTitle}>⚕ SIMULADOR DE MESA</div>
                  <div style={{ fontSize: 12, color: C.muted, marginBottom: 18, lineHeight: 1.7 }}>
                    Selecciona un procedimiento para ver la lista de insumos de mesa y practicar la preparación de pabellón.
                  </div>
                  {specKeys.map(key => {
                    const sp = data.procedures[key];
                    return (
                      <div key={key} style={{ marginBottom: 14 }}>
                        <div style={{ fontSize: 10, color: sp.color, letterSpacing: 3, marginBottom: 8 }}>{sp.icon} {sp.label.toUpperCase()}</div>
                        <div style={{ display: "flex", flexDirection: "column", gap: 6 }}>
                          {sp.procedures.map(p => (
                            <div key={p.id} style={{ ...S.card, cursor: "pointer", padding: "12px 16px" }}
                              onClick={() => openProc(key, p)}>
                              <div style={{ fontSize: 13, color: "#c8dce8", fontWeight: 700 }}>{p.name}</div>
                              <div style={{ fontSize: 11, color: C.muted, marginTop: 3 }}>{(p.supplies || []).length} insumos registrados</div>
                            </div>
                          ))}
                        </div>
                      </div>
                    );
                  })}
                </>
              )}

              {/* SETTINGS */}
              {sideTab === "settings" && (
                <>
                  <div style={S.panelTitle}>⚙️ CONFIGURACIÓN</div>
                  {[
                    { label: "Cambiar Contraseña", fields: ["Contraseña actual", "Nueva contraseña", "Confirmar nueva contraseña"], type: "password" },
                    { label: "Cambiar Teléfono", fields: ["+56 9 XXXX XXXX"], type: "tel" },
                    { label: "Cambiar Correo", fields: ["nuevo@correo.cl", "Confirmar correo"], type: "email" },
                  ].map((section, i) => (
                    <div key={i} style={{ ...S.card, marginBottom: 12 }}>
                      <div style={S.cardTitle}><span>{section.label.toUpperCase()}</span></div>
                      <div style={{ display: "flex", flexDirection: "column", gap: 8 }}>
                        {section.fields.map((f, j) => <input key={j} type={section.type} style={S.input} placeholder={f} />)}
                        <button style={S.btn(C.accent)} onClick={() => setSaveMsg("Actualizado ✓")}>ACTUALIZAR</button>
                      </div>
                    </div>
                  ))}
                  <div style={{ ...S.card, borderColor: "#1a0a0a" }}>
                    <div style={S.cardTitle}><span>DATOS DE LA APLICACIÓN</span></div>
                    <div style={{ fontSize: 11, color: C.muted, lineHeight: 1.8, marginBottom: 12 }}>
                      Procedimientos: {specKeys.reduce((a, k) => a + data.procedures[k].procedures.length, 0)}<br />
                      Términos en diccionario: {data.dictionary.length}<br />
                      Preguntas frecuentes: {data.faqs.length}
                    </div>
                    <button style={S.btn(C.red)} onClick={resetToDefault}>RESTAURAR AL MANUAL ORIGINAL</button>
                  </div>
                </>
              )}
            </>
          )}
        </main>
      </div>
    </div>
  );
}
